from __future__ import print_function
import logging

from . import vfile

try:
    import h5py
    HAS_H5PY = True
except ImportError:
    HAS_H5PY = False
    logging.warn("h5py library not found, no hdf5 support.")



[docs]class H5pyDataWrapper(vfile.DataWrapper): #pylint: disable=R0924
    """  """
    fname = None
    loc = None

    def __init__(self, fname, loc):
        super(H5pyDataWrapper, self).__init__()
        self.fname = fname
        self.loc = loc

        with h5py.File(self.fname, 'r') as f:
            dset = f[loc]
            self._shape = dset.shape
            self._dtype = dset.dtype
